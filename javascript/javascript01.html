01. 변수 : 데이터 저장

변수는 데이터를 저장하는 "저장소" 이며 이 저장소에는 숫자, 문자, 함수, 객체 등을 저장할수 있습니다.

{
    var x = 100;    //변수 X에 숫자 100을 저장 합니다.        
    var y = 200;    //변수 y에 숫자 200을 저장 합니다.            
    var z = "javascript";     //변수 z에 문자 "javascript"를 저장 합니다.

    console.log(x); //x값을 콘솔에 출력한다.
    console.log(y); //y값을 콘솔에 출력한다.
    console.log(z); //z값을 콘솔에 출력한다.
}
결과 확인하기
100
200
"javascript"

02. 변수 : 데이터 저장 + 데이터 변경

변수는 데이터를 저장 할수도 있지만 변경도 가능 합니다. 

{
    let x = 100;    //변수 x에 100을 저장한다.
    let y = 200;    //변수 y에 200을 저장한다.
    let z = "javascript";   //변수 z에 javascript를 저장한다.

    x = 300;    //변수 X의 값인 100이 300으로 변경됬다.
    y = 400;    //변수 y의 값인 200이 400으로 변경됬다.
    z = "react";    //변수 z의 값인 javascript가 react로 변경됬다.

    console.log(x);     //콘솔에 x의 값을 출력한다.
    console.log(y);     //콘솔에 y의 값을 출력한다.
    console.log(z);     //콘솔에 z의 값을 출력한다.
}
결과 확인하기
300
400
"react"

03. 변수 : 데이터 저장 + 데이터 변경 + 데이터 추가

변수는 데이터를 저장,변경할수도있고 추가도 할 수 있습니다.


{
    let x = 100;
    let y = 200;
    let z = "javascript";

    x += 300;   // 변수 x의 값인 100에 300을 추가 (x+300)
    y += 400;   // 변수 y의 값인 200에 400을 추가 (y+400)
    z += "react"; // 변수 z의 값인 javascript에 react를 추가 (z+react)

    console.log(x);     //콘솔에 x의 값을 출력한다.
    console.log(y);     //콘솔에 y의 값을 출력한다.
    console.log(z);     //콘솔에 z의 값을 출력한다.
}
결과 확인하기
400
600
"javascriptreact"

04. 상수 : 데이터 저장 + 데이터 변경(X)

상수는 데이터 저장은 가능하나 변경은 불가능합니다.
상수(const)는 이미 선언한 상수에 대해 중복 선언 불가능하며 상수의 값을 재지정할 수도 없습니다. 

{
    const x = 100;      //상수 x의 값에 100을 저장합니다.
    const y = 200;      //상수 y의 값에 200을 저장합니다.
    const z = "javascript";     //상수 z의 값에 javascript를 저장합니다.

    x = 300;    //상수 x의 값을 300으로 변경한다.
    y = 400;    //상수 y의 값을 400으로 변경한다.
    z = "react";    //상수 y의 값을 react로 변경한다.

    console.log(x);     //콘솔에 x의 값을 출력한다.
    console.log(y);     //콘솔에 y의 값을 출력한다.
    console.log(z);     //콘솔에 z의 값을 출력한다.
}
